    <pre><code class=" javascript">var myDialog = art.dialog();// 初始化一个带有loading图标的空对话框

	url: 'http://web5.qq.com/content?id=1',
    success: function (data) {
        myDialog.content(data);// 填充对话框内容
    }
});
</code></pre>
